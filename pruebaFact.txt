program factorial;
var int i, num, fact;

func int patitoPelos(int a, int b){
    return(a*b);
}

func int funFact(int num){
    from i=1 to num + 1 do{
        fact = patitoPelos(i, fact);
    }
    return(fact);
}

main(){
    print("Ingresa el numero a obtener el factorial");
    read(num);
    fact = 1;
    print("El factorial es", funFact(num));
}
